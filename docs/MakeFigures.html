plot_umap <- function(seurat.object,
                      group, 
                      title,
                      save.dir = output,
                      cols = NULL,
                      project.name, 
                      size = 10){
  #with legend
  umap <- DimPlot(seurat.object,
                  group.by = group,
                  cols = cols,
                  shuffle = T, 
                  pt.size = 1,
                  label = F) + NoAxes() + ggtitle(title)
  
  umapFileName <- sprintf('%s/%s_umap_%s.pdf', save.dir, project.name, group)
  ggsave(umapFileName,
         umap,
         device="pdf",
         width=size,
         height=size,
         units="cm",
         scale=1)
  
  #without legend
  umap.nolegend <- umap + NoLegend() + NoAxes() + ggtitle('')
  umapFileName <- sprintf('%s/%s_umap_%s_nolegend.pdf', save.dir, project.name, group)
  ggsave(umapFileName,
         umap.nolegend,
         device="pdf",
         width=size,
         height=size,
         units="cm",
         scale=1)
  
  return(umap)
}
